    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const preloader = document.getElementById("preloader");

            // fast hide
            setTimeout(() => {
                preloader.classList.add("hidden");
            }, 400);

            // safety fallback (just in case)
            setTimeout(() => {
                preloader.style.display = "none";
            }, 1500);

            // 🔊 SOUND
            function beep(freq = 500, duration = 150) {
                const ctx = new (window.AudioContext || window.webkitAudioContext)();
                const osc = ctx.createOscillator();
                const gain = ctx.createGain();

                osc.type = "sine";
                osc.frequency.value = freq;

                osc.connect(gain);
                gain.connect(ctx.destination);

                osc.start();

                gain.gain.setValueAtTime(1, ctx.currentTime);
                gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + duration / 1000);

                setTimeout(() => osc.stop(), duration);
            }

            // 🎯 ELEMENTS
            const apiKeyInput = document.getElementById('apiKey');
            const generateBtn = document.getElementById('generateBtn');
            const planOutput = document.getElementById('planOutput');
            const refineBtn = document.getElementById('refineBtn');
            const refineInput = document.getElementById('refineInput');
            const regenBtn = document.getElementById('regenBtn');
            const addSubjectBtn = document.getElementById("addSubjectBtn");
            const subjectContainer = document.getElementById("subject-container");

            addSubjectBtn.addEventListener("click", () => {
                beep(600, 80); // 🔊 sound feedback

                const newSubject = document.createElement("div");
                newSubject.className = "subject-entry glass-panel p-4 !bg-white/5 !shadow-none !border-white/10 animate-fade-in";

                newSubject.innerHTML = `
    <div class="flex gap-3 mb-3">
      <input type="text" placeholder="Subject"
        class="sub-name glass-input flex-1 p-3 text-sm font-semibold">
      <select class="sub-lvl glass-input w-24 p-3 text-xs font-bold text-cyan-300 text-center">
        <option value="1">Beginner</option>
        <option value="2">Intermediate</option>
        <option value="3" selected>Advanced</option>
      </select>
    </div>
    <textarea placeholder="Tailoring Context"
      class="sub-tailor glass-input w-full p-3 text-xs h-16 resize-none"></textarea>

    <button class="remove-btn mt-2 text-xs text-red-400 hover:text-red-300">
      ✖ Remove
    </button>
  `;

                // 🔥 REMOVE FUNCTION
                newSubject.querySelector(".remove-btn").addEventListener("click", () => {
                    newSubject.remove();
                });

                subjectContainer.appendChild(newSubject);
            });

            // 🔁 REGENERATE
            regenBtn.addEventListener("click", () => generateBtn.click());
           

            // 🧠 TIMER SYSTEM
            let currentTimerDuration = 25 * 60;
            let remainingTime = currentTimerDuration;
            let timerInterval = null;

            function updateTimerDisplay() {
                const m = Math.floor(remainingTime / 60);
                const s = remainingTime % 60;
                document.getElementById("timerDisplay").innerText =
                    `${m}:${s.toString().padStart(2, '0')}`;
            }

            function setTimerFromPlan(minutes) {
                currentTimerDuration = minutes * 60;
                remainingTime = currentTimerDuration;
                clearInterval(timerInterval);
                updateTimerDisplay();
            }

            function startTimer() {
                clearInterval(timerInterval);

                timerInterval = setInterval(() => {
                    if (remainingTime <= 0) {
                        clearInterval(timerInterval);
                        beep(300, 300);
                        document.getElementById("timerSubjectLabel").innerText = "Session Complete!";
                        return;
                    }

                    remainingTime--;
                    updateTimerDisplay();
                }, 1000);
            }

            document.getElementById("timerToggle").addEventListener("click", () => {
                beep(800, 100);
                startTimer();
            });

            document.getElementById("timerReset").addEventListener("click", () => {
                beep(500, 80);
                setTimerFromPlan(currentTimerDuration / 60);
            });

            document.querySelectorAll(".timeAdd").forEach(btn => {
                btn.addEventListener("click", () => {
                    const add = parseInt(btn.dataset.add, 10);
                    remainingTime += add;
                    updateTimerDisplay();
                    beep(900, 80);
                });
            });

            // 🤖 MODELS
            async function callAI(system, user) {
                const geminiKey = apiKeyInput.value.trim();
                // OpenAI and Anthropic support disabled for now (reserved for future upgrade)
                // const openaiKey = document.getElementById("openaiKey")?.value?.trim();
                // const anthropicKey = document.getElementById("anthropicKey")?.value?.trim();

                // 🥇 GEMINI (FAST)
                if (geminiKey) {
                    const endpoint = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash:generateContent?key=${encodeURIComponent(geminiKey)}`;

                    const payload = {
                        contents: [{
                            role: "user",
                            parts: [{ text: system + "\n\nUser Request: " + user }]
                        }],
                        generationConfig: {
                            responseMimeType: "application/json"
                        }
                    };

                    const res = await fetch(endpoint, {
                        method: "POST",
                        headers: { "Content-Type": "application/json" },
                        body: JSON.stringify(payload)
                    });

                    const data = await res.json();
                    const text = data?.candidates?.[0]?.content?.parts?.[0]?.text;
                    try {
                        return {
                            data: JSON.parse(text),
                            modelUsed: "Gemini Flash"
                        };
                    } catch (e) {
                        throw new Error("Invalid Gemini response JSON: " + (e.message || e));
                    }
                }

                // OpenAI & Anthropic branches removed/disabled to simplify current flow.
                throw new Error("OpenAI and Anthropic support are temporarily disabled. Provide a Gemini API key to use the engine.");
            }

            // 🎨 RENDER
            function renderCards(data) {
                planOutput.innerHTML = "";

                if (!Array.isArray(data)) {
                    planOutput.innerHTML = `
          <div class="p-4 bg-yellow-500/10 border border-yellow-500 text-yellow-100 rounded-xl">
            Unexpected AI format — expected an array.
          </div>
        `;
                    return;
                }

                data.forEach(item => {
                    const card = document.createElement("div");

                    const colorClass = item.color || "border-emerald-500";
                    card.className = `glass-panel p-6 border-l-4 ${colorClass} mb-5 cursor-pointer hover:scale-[1.02] transition`;

                    card.innerHTML = `
        <h3 class="text-xl font-bold">${item.name || "Untitled"}</h3>
        <p class="text-sm text-slate-400">${item.strategy || ""}</p>
        <p class="text-xs text-cyan-300 mt-2">${item.aiInsight || ""}</p>
        <div class="text-right text-lg font-bold mt-3">${item.time || 0} min</div>
      `;

                    card.addEventListener("click", () => {
                        beep(600, 80);
                        setTimerFromPlan(Number(item.time) || 25);
                        document.getElementById("timerSubjectLabel").innerText = item.name || "Focus";
                    });

                    planOutput.appendChild(card);
                });
            }

            // 🚀 GENERATE
            
            generateBtn.addEventListener('click', async () => {
                beep(700, 100);

                const entries = document.querySelectorAll('.subject-entry');
                let parsedSubjects = [];

                entries.forEach(entry => {
                    const name = (entry.querySelector('.sub-name')?.value || "").trim();
                    const lvl = entry.querySelector('.sub-lvl')?.value || "";
                    const tailor = (entry.querySelector('.sub-tailor')?.value || "").trim();

                    if (name) {
                        parsedSubjects.push(`Subject: ${name}, Level: ${lvl}, Context: ${tailor}`);
                    }
                });

                if (parsedSubjects.length === 0) {
                    planOutput.innerHTML = `
        <div class="text-center py-20 text-slate-400">
          <p class="text-lg font-semibold">No subjects added</p>
          <p class="text-sm mt-2">Add subjects to generate your AI plan</p>
        </div>
      `;
                    return;
                }

                planOutput.innerHTML = `
      <div class="flex flex-col items-center justify-center py-20">
        <div class="w-10 h-10 border-4 border-cyan-400 border-t-transparent rounded-full animate-spin"></div>
        <p class="mt-4 text-cyan-300 text-sm">Generating AI Plan...</p>
      </div>
    `;

                const systemInstruction = `
Generate a study plan as a JSON array.

Schema:
[
  {
    "name": "string",
    "time": number,
    "mode": "string",
    "strategy": "string",
    "aiInsight": "string",
    "color": "border-emerald-500 | border-amber-500 | border-rose-500"
  }
]
`;

                try {
                    const result = await callAI(
                        systemInstruction,
                        parsedSubjects.join(" | ")
                    );

                    // Log which model was used
                    console.log("Using AI:", result.modelUsed);

                    renderCards(result.data);

                    planOutput.insertAdjacentHTML(
                        "afterbegin",
                        `<div class="text-[10px] text-cyan-400 mb-4">Model Used: ${result.modelUsed}</div>`
                    );

                } catch (err) {
                    planOutput.innerHTML = `
        <div class="p-4 bg-red-500/20 border border-red-500 text-red-200 rounded-xl">
          API Error: ${err.message}
        </div>
      `;
                    console.error(err);
                }
            });
            generateBtn.disabled = false;
            // 🔁 REFINE
            refineBtn.addEventListener('click', async () => {
                const req = refineInput.value.trim();
                if (!req) return;

                try {
                    const result = await callAI(
                        "Update this study plan JSON",
                        req
                    );
                    console.log("Using AI (refine):", result.modelUsed);
                    renderCards(result.data);
                    refineInput.value = "";
                } catch (err) {
                    alert(err.message);
                    console.error(err);
                }
            });
          
        });
